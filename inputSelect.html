InputSelect 使用演示

在自定义类型输入框中，开发人员可为用户预设可选值(下拉菜单内容)，用户也可自定义取值(通过输入框进行输入)

生成单个输入框 
// html 代码
<div id="select-test"> </div>

// js 代码
<script>
/*下拉列表接口*/
//seeAsTrans==false,下拉表中显示opt2.val，点击给输入框赋值option2(点击显示不一致)，
//给隐藏输入框赋值option2
//seeAsTrans==true,下拉表中显示opt2.val，点击给输入框赋值opt2.val(点击显示一致)，
//给隐藏输入框赋值option2
    
$('#select-test').toSelect({  
  "initVal": "select", //初始化时显示值
  "editable": "1",    //editable: 是否可输入  0：禁用输入， 1： 可输入
  "seeAsTrans": true,//不写默认为false
  "options":[{
    "option1":"option1",  //下拉列表参数
    "option2":"opt2.val",
    "6-12": "上午",//seeAsTrans作用演示，用户点击上午时，隐藏输入框向后台传6-12
    "option3":"", //键值对中，值为空时，应用键名作为值
    ".divider":".divider",   //分割线
    ".hand-set":"手动设定"
  }]
});
</script>

生成一组输入框 
// html 代码
<div class="select-classbased"></div>
<div class="select-classbased"></div>
<div class="select-classbased"></div>

// js 代码
<script>
  $('.select-classbased').toSelect({
    "initVal": "by class",
    "editable": "1",
    "options":[{
      "option1":"option1",
      "option2":"option2",
      "option3":"",
      ".divider":".divider",
      ".hand-set":"手动设定"
    },
    //可以为每个输入框指定不同的选项表，如果指定数少于预期，选项表将会被循环（交替）使用
    {
      "option1":"hi, let's rock&roll",
      "option2":"la bonjour",
      ".divider":".divider",
      ".hand-set":"手动设定"
    }]
  });
</script>

给选项表添加或者删除选项 
// html 代码								
<div class="select-classbased"></div>

// js 代码
<script>
  //示例中的输入框与byclass生成的第一个原来完全相同
  //删除第二个选项
  $('.select-classbased')[3].removeLi(1);
  //在末尾添加一个新选项（目前只能在末尾添加）
  $('.select-classbased')[3].appendLi({"new":"New York"});
</script>
